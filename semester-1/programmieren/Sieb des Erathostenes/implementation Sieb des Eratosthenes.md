int sieve()

    for (int i = 2; i < UPPER_LIMIT; i++)

void sieve()
{
    for (int i = 3; i * i < UPPER_LIMIT; i += 2)
    {
        if (numbers[i])
        {
            for (int j = i * i; j < UPPER_LIMIT; j += 2 * i)
            {
                numbers[j] = 0;
            }
        }
    }
}
